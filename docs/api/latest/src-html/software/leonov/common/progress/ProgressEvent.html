001package software.leonov.common.progress;
002
003import java.util.Optional;
004
005/**
006 * A {@code ProgressEvent} is used to notify {@link ProgressListener}s that the progress has changed.
007 * 
008 * @author Zhenya Leonov
009 */
010public interface ProgressEvent {
011
012    /**
013     * Returns the current progress.
014     * 
015     * @return the current progress
016     */
017    public long getProgress();
018
019    /**
020     * Returns the maximum value.
021     * 
022     * @return the maximum value
023     */
024    public Optional<Long> getMaximum();
025
026}
