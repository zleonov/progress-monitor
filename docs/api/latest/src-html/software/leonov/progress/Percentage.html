001package software.leonov.progress;
002
003import static java.util.Objects.requireNonNull;
004
005import java.text.DecimalFormat;
006import java.text.NumberFormat;
007import java.text.ParseException;
008import java.util.Locale;
009
010/**
011 * A skeletal implementation of the {@code ProgressListener} interface which calculates the
012 * {@link ProgressMonitor#getProgress() progress} value as a percentage of the {@link ProgressMonitor#getMaximum()
013 * maximum} value.
014 * <p>
015 * Extending class need only to implement the {@link #onUpdate(String)} method, which receives the formatted
016 * representation of the percent value. The format pattern can be specified using {@link #Percentage(DecimalFormat)} or
017 * {@link #Percentage(Locale, String)} methods. By default, the this class formats the value to the nearest whole
018 * percent without decimal places. Users of this class are guaranteed never to receive duplicate percent values and will
019 * never encounter zero as an input.
020 * <p>
021 * Using this listener when the maximum value is not {@link ProgressMonitor#setMaximum(long) specified} will lead to an
022 * {@code IllegalStateException}.
023 * <p>
024 * For example: <pre><code class="line-numbers match-braces language-java">
025 *     ...
026 *     final ProgressMonitor progress = new ProgressMonitor().addProgressListener(new Percentage() {
027 *
028 *         {@literal @}Override
029 *         public void onUpdate(final String pct) {
030 *             System.out.println("Processed " + pct + "%");
031 *         }
032 *     });
033 *     ...
034 * </code></pre>
035 *
036 * @author Zhenya Leonov
037 */
038public abstract class Percentage implements ProgressListener {
039
040    private final DecimalFormat format;
041    private String              last;
042
043    /**
044     * Creates a new {@link Percentage} {@code ProgressListener}.
045     */
046    public Percentage() {
047        format = new DecimalFormat("#");
048    }
049
050    /**
051     * Creates a new {@link Percentage} {@code ProgressListener} which uses the specified {@code DecimalFormat} to format
052     * the percent value.
053     * 
054     * @param format the format to use
055     */
056    public Percentage(final DecimalFormat format) {
057        requireNonNull("format", "format == null");
058        this.format = format;
059    }
060
061    /**
062     * Creates a new {@link Percentage} {@code ProgressListener} which uses the specified {@code Locale} and {@code pattern}
063     * to format the percent value.
064     * 
065     * @param locale  the {@code Locale} to use when formatting the percent value
066     * @param pattern the pattern to apply
067     */
068    public Percentage(final Locale locale, final String pattern) {
069        requireNonNull("locale", "locale == null");
070        requireNonNull("pattern", "pattern == null");
071        this.format = (DecimalFormat) NumberFormat.getNumberInstance(locale);
072        this.format.applyPattern(pattern);
073    }
074
075    /**
076     * {@inheritDoc} Then {@link DecimalFormat#format(double) formats} the value as a percentage of the
077     * {@link ProgressMonitor#getMaximum() maximum} value and invokes {@link #onUpdate(String)}.
078     */
079    @Override
080    public final void progressChanged(final ProgressEvent event) {
081        final double percentage = (double) event.getProgress() / event.getMaximum().orElseThrow(() -> new IllegalStateException("maximum value undefined")) * 100;
082        final String pct        = format.format(percentage);
083
084        try {
085            if (!pct.equals(last) && format.parse(pct).doubleValue() != 0) {
086                last = pct;
087                onUpdate(pct);
088            }
089        } catch (final ParseException e) {
090            // cannot happen
091        }
092    }
093
094    /**
095     * Receives the formatted percent value as a calculation of the {@link ProgressMonitor#getProgress() progress} against
096     * the {@link ProgressMonitor#getMaximum() maximum} value. Users are guaranteed never to receive duplicate percent
097     * values and will never encounter zero as an input.
098     * 
099     * @param pct the formatted percent value
100     * @throws IllegalStateException if the maximum value is not {@link ProgressMonitor#setMaximum(long) specified}
101     */
102    public abstract void onUpdate(final String pct);
103
104}
